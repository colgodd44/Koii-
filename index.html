// 1. Audio Engine (Synths) const kick = new Tone.MembraneSynth({ volume: 5 }).toDestination(); const snare = new Tone.NoiseSynth({ volume: -5, envelope: { decay: 0.1 } }).toDestination(); const hihat = new Tone.MetalSynth({ volume: -10, envelope: { decay: 0.05 } }).toDestination(); const perc = new Tone.DuoSynth({ volume: -10 }).toDestination(); // 2. Sequencer State let isRecording = false; let loopStarted = false; const sequence = new Tone.Part((time, event) => { hit(event.type, true, time); }, []).start(0); sequence.loop = true; sequence.loopEnd = "1n"; // 1 bar loop async function boot() { await Tone.start(); Tone.Transport.bpm.value = 120; document.getElementById('display').innerText = "120.0 BPM"; document.getElementById('pwrBtn').style.background = "#2ea44f"; document.getElementById('pwrBtn').innerText = "SYSTEM ONLINE"; } function hit(type, isPlayback = false, time = Tone.now()) { if (Tone.context.state !== 'running') return; // Play the sound if (type === 'kick') kick.triggerAttackRelease("C1", "8n", time); if (type === 'snare') snare.triggerAttackRelease("8n", time); if (type === 'hihat') hihat.triggerAttackRelease("32n", time); if (type === 'perc') perc.triggerAttackRelease("G4", "32n", time); // If we are recording and this is a manual tap, save it if (isRecording && !isPlayback) { const playhead = Tone.Transport.seconds % Tone.Time("1n").toSeconds(); sequence.add(playhead, { type: type }); } if (!isPlayback) { document.getElementById('display').innerText = type.toUpperCase(); if (navigator.vibrate) navigator.vibrate(15); } } function togglePlay() { if (!loopStarted) { Tone.Transport.start(); loopStarted = true; document.getElementById('display').innerText = "PLAYING"; } else { Tone.Transport.stop(); loopStarted = false; document.getElementById('display').innerText = "STOPPED"; } } function toggleRec() { isRecording = !isRecording; document.getElementById('display').innerText = isRecording ? "RECORDING" : "REC OFF"; document.getElementById('display').style.color = isRecording ? "#ff0000" : "#ff5f00"; }
